Part 7: How to Run

Prerequisites

sudo apt update && sudo apt install -y aircrack-ng iw wireless-tools docker.io python3-pip
cd /home/ubuntu/NTTH/backend && pip install -r requirements.txt
sudo airmon-ng check kill
sudo airmon-ng start wlx24ec99bfe292

Development Mode

cd /home/ubuntu/NTTH/backend
sudo uvicorn app.main:app --reload --port 8000 --log-level info

Verification

curl http://localhost:8000/api/v1/system/health

TOKEN=$(curl -s -X POST http://localhost:8000/api/v1/auth/login \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","password":"NtthAdmin2026"}' | jq -r .access_token)

curl -H "Authorization: Bearer $TOKEN" http://localhost:8000/api/v1/wireless/devices

Attack Testing

# Deauth attack test
sudo aireplay-ng --deauth 20 -a <TARGET_BSSID> wlan0mon

# Rogue AP test
sudo hostapd /tmp/rogue_ap.conf
⚠️ Docker note: Container needs --privileged or cap_add: [NET_ADMIN, NET_RAW] for AR9271 access.
